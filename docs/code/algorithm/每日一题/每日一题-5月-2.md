function validIPv4(arr) {
    const reg = /^0$|^[1-9][0-9]{0,2}$/
    if (arr.every(item => reg.test(item) && item < 256)) {
        return 'IPv4'
    }
    return 'Neither'
}

function validIPv6(arr) {
    const reg = /^[0-9a-fA-F]{1,4}$/
    if (arr.every(item => reg.test(item))) {
        return 'IPv6'
    }
    return 'Neither'
}
